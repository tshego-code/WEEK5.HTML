  <!-- Part 1: Basics -->

  <input type="number" id="ageInput" placeholder="Enter your age">
  <button id="checkAgeBtn">Check Age</button>
  <p id="ageResult"></p>

  <!-- Part 2: Functions -->
  <h2>Part 2: Functions</h2>
  <p>Click to calculate the total price of an item with VAT included:</p>
  <button onclick="calculateTotal(200, 15)">Calculate Total Price</button>
  <p id="totalOutput"></p>

  <!-- Part 3: Loops -->
  <h2>Part 3: Loops</h2>
  <p>Click to show numbers 1 to 5 generated with a loop:</p>
  <button id="loopBtn">Show Numbers 1-5</button>
  <ul id="numberList"></ul>

  <!-- Part 4: DOM Manipulation -->
  <h2>Part 4: DOM Manipulation</h2>
  <p>Play around with buttons to see the DOM in action:</p>

  <button id="addItemBtn">Add List Item ➕</button>
  <ul id="dynamicList"></ul>

  <script>
    // 🎯 PART 1: VARIABLES & CONDITIONALS
    const checkAgeBtn = document.getElementById("checkAgeBtn");
    const ageInput = document.getElementById("ageInput");
    const ageResult = document.getElementById("ageResult");

    checkAgeBtn.addEventListener("click", () => {
      let age = parseInt(ageInput.value);

      if (isNaN(age)) {
        ageResult.textContent = "⚠️ Please enter a valid number.";
      } else if (age >= 18) {
        ageResult.textContent = "✅ You are an adult.";
      } else {
        ageResult.textContent = "👶 You are a minor.";
      }
    });

    // ❤️ PART 2: FUNCTIONS
    // Function to calculate total price with VAT
    function calculateTotal(price, vatPercent) {
      let total = price + (price * vatPercent / 100);
      document.getElementById("totalOutput").textContent = 
        `Total price (with VAT): R${total} (calculated by Tshegofatso Moloto)`;
      return total;
    }

    // A simple function to greet the user
    function greetUser(name) {
      return `Hello, ${name}! This is Tshegofatso Moloto’s JavaScript demo 🚀`;
    }
    console.log(greetUser("Lecturer / Viewer")); // Personal greeting in console

    // 🔁 PART 3: LOOPS
    const loopBtn = document.getElementById("loopBtn");
    const numberList = document.getElementById("numberList");

    loopBtn.addEventListener("click", () => {
      numberList.innerHTML = "";
      for (let i = 1; i <= 5; i++) {
        let li = document.createElement("li");
        li.textContent = `Number ${i} (added by Tshegofatso Moloto)`;
        numberList.appendChild(li);
      }
    });

    // Example of a while loop - countdown in console
    let countdown = 5;
    while (countdown > 0) {
      console.log("Countdown (" + countdown + ") logged by Tshegofatso Moloto");
      countdown--;
    }

    // 🌐 PART 4: DOM MANIPULATION
    const toggleBtn = document.getElementById("toggleBtn");
    const addItemBtn = document.getElementById("addItemBtn");
    const dynamicList = document.getElementById("dynamicList");

    // Toggle dark/light mode
    toggleBtn.addEventListener("click", () => {
      document.body.classList.toggle("dark-mode");
    });

    // Add list items dynamically
    let itemCount = 0;
    addItemBtn.addEventListener("click", () => {
      itemCount++;
      let li = document.createElement("li");
      li.textContent = `Dynamic Item ${itemCount} (added by Tshegofatso Moloto)`;
      dynamicList.appendChild(li);
    });

    // Update intro text on page load
    document.getElementById("intro").textContent = 
      "Hi 👋 I’m Tshegofatso Moloto, a Software Engineering student at PLP Academy. This is my JavaScript assignment where I bring a static page to life 🎉";
  </script>
